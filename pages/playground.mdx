<iframe
  src="https://playground-47d69.ondigitalocean.app?embed=true&embed_options=hide_loading_screen&embed_options=disable_scrolling&log="
  width="100%"
  style={{
    height: "1600px",
    filter: "brightness(0.96875)",
    borderRadius: "8px",
  }}
  id="playgroundA"
  onLoad={(event) => {
    let srcUpdated = false;

    if (window.visiblePlayground === "B") {
      setTimeout(() => {
        document.getElementById("playgroundA").style.opacity = "0";
        document.getElementById("playgroundA").style.pointerEvents = "none";
        document.getElementById("playgroundB").style.opacity = "1";
        document.getElementById("playgroundB").style.pointerEvents = "initial";
      }, 1000);
    } else {
      setTimeout(() => {
        document.getElementById("playgroundB").style.opacity = "0";
        document.getElementById("playgroundB").style.pointerEvents = "none";
        document.getElementById("playgroundA").style.opacity = "1";
        document.getElementById("playgroundA").style.pointerEvents = "initial";
      }, 1000);

      window.addEventListener("message", (_event) => {
        if (
          typeof _event.data === "string" &&
          _event.data.indexOf("nxtl_") === 0 &&
          !srcUpdated
        ) {
          const curr = document.getElementById("playgroundA"),
            next = document.getElementById("playgroundB");

          const currentSrc = new URL(curr.src);
          const nextSrc = new URL(next.src);
          nextSrc.searchParams.set(
            "log",
            `${currentSrc.searchParams.get("log")}\n${
              _event.data.split("nxtl_")[1]
            }`
          );
          window.visiblePlayground = "B";
          next.src = nextSrc.toString();
          srcUpdated = true;
        }
      });
    }

}}
/>
